#include <iostream>
#include <string>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Russian");
	string h;
	cout << "Введите строчище: ";
	getline(cin, h);
	for (int i = 0; i < h.length(); i++)
	{
		if (isdigit(h[i]))
		{
			h[i] = ' ';
		}
	}
	cout << "Результатище - " << h << endl;
	system("pause");
}
