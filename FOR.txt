#include <iostream>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Rus");
	int n;
	cout<<"Введите число"<<endl;
	cin>>n;
	if(n<=0)
	{
		cout<<"Ошибка"<<endl;
	}

	cout<<"Таблица умножения на число "<<n<<endl;

	for(int i = 1 ; i <= 10; i = i + 1)
	{
     cout<<i<<"*"<<n<<"="<<i*n<<endl;
	}

	system("pause");
}
