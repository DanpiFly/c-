#include <iostream>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Rus");
	float n;
	float g;
	int count = 0;
	cout<<"Введите первое число" << endl;
	cin >> n;
	cout << "Введите второе число" << endl;
	cin >> g;
	cout << "Выберите действие 1 - сложение, 2 - вычитание, 3 - умножение, 4 - деление" << endl;
	cin >> count;
	switch (count)
	{
	case 1:
	{
		cout << n << " + " << g << " = " << n + g << endl;
		break;
	}
	case 2:
	{
		cout <<n << " - " << g << " = " << n - g << endl;
		break;
	}
	case 3:
	{
		cout << n << " * " << g << " = " << n * g << endl;
		break;
	}
	case 4:
	{
	cout << n << " / " << g << " = " << n / g << endl;
	break;
	}
	default:
	{
		cout << "Ошибка" << endl;
	}

	}



	system("pause");
}
